## Task 2: traceProjection 투영 모듈 (TDD)

**Files:** `frontend/src/features/trace/traceProjection.ts`, `frontend/src/features/trace/traceProjection.test.ts`

### 출력 타입
```ts
import type { TraceNode, TraceEdge, TraceRing } from '../pathTrace/types';
export interface TraceStep { id: string; label: string; isEndpoint: boolean; isFiberEdge: boolean }
export interface TraceProjection {
  seedCableId: string;
  nodeIds: string[];      // 하이라이트 — cableTrace 그대로(접기 전, 실제 자산/슬롯)
  cableIds: string[];     // 하이라이트 — cableTrace 그대로
  steps: TraceStep[];     // 경로상세 — 시드에서 정렬된 메인 체인(슬롯→OFD 접힘)
  nodes: TraceNode[];     // 토폴로지 — OFD-접기 노드(설비=모듈, OFD=EQP-OFD)
  edges: TraceEdge[];     // 토폴로지 — 광edge(OFD↔OFD, type 'fiberPath', #K) + cable edge
  rings: TraceRing[];     // 토폴로지 — 광 edge 위 detectRings
  seedFiberEdgeId: string | null; // 토폴로지 seed-ring 강조용(시드에 인접한 OPGW edge id)
  truncated: boolean;
}
```

### 규칙
- start = 시드 케이블 끝점 중 passive(connectionKind null) 쪽; 둘 다 passive/특수면 sourceAssetId.
- `cableType` = 시드의 cableType. `{nodeIds, cableIds, truncated}` = `cableTrace(start, cableType, graph.assets, graph.cables)`.
- collapse(n): `kindOf(n)==='conduit'` 이고 부모 있으면 `parentById.get(n)`, 아니면 n.
- 코어번호 K: 시드 `number` ?? 트레이스 내 OUT 역할 케이블의 number.
- 토폴로지 nodes: 트레이스 노드를 collapse 후 유니크. 각 노드 TraceNode{ nodeId, nodeName(접힌 OFD 면 그 OFD 이름), substationName(subNameById), substationId:'', floorId:null, materialCategoryCode(code==='OFD'?'EQP-OFD':null), isSource(접힌 start), isTarget:false }.
- 토폴로지 edges: 각 트레이스 케이블 →
  - OPGW(role IN-IN): `{id:c.id, sourceAssetId:collapse(src), targetAssetId:collapse(tgt), type:'fiberPath', fiberPathId:c.id, fiberPortNumber:K}`.
  - 그 외(OUT 등): `{id:c.id, sourceAssetId:collapse(src), targetAssetId:collapse(tgt), type:'cable'}` (src==tgt 면 생략).
- rings: 광(type 'fiberPath') edge 만으로 adjacency/edgeMap/nodeMap 구성 → `detectRings`.
- seedFiberEdgeId: 시드가 OUT 이면 그 슬롯에 붙은 OPGW edge id; 시드가 OPGW 면 자신.
- steps: 트레이스 서브그래프(접기 전 케이블)에서 start→최원거리 노드 BFS 체인. 체인을 걷되 연속 슬롯→OFD 는 한 노드로 접고, 들어온 edge 가 OPGW 면 isFiberEdge. label = OFD 면 substationName, 설비면 nameById. isEndpoint = start 또는 체인 끝.

### Step 3: 구현 — Create `traceProjection.ts`:
```ts
import { cableTrace } from './cableTrace';
import type { TraceGraph } from './traceGraph';
import { detectRings } from '../../utils/graph/cycleDetection';
import type { TraceNode, TraceEdge, TraceRing } from '../pathTrace/types';

export interface TraceStep { id: string; label: string; isEndpoint: boolean; isFiberEdge: boolean }
export interface TraceProjection {
  seedCableId: string;
  nodeIds: string[];
  cableIds: string[];
  steps: TraceStep[];
  nodes: TraceNode[];
  edges: TraceEdge[];
  rings: TraceRing[];
  seedFiberEdgeId: string | null;
  truncated: boolean;
}

const roleAt = (c: { sourceAssetId?: string | null; targetAssetId?: string | null; sourceRole?: string | null; targetRole?: string | null }, a: string) =>
  (c.sourceAssetId === a ? c.sourceRole : c.targetRole) ?? null;
const otherEnd = (c: { sourceAssetId?: string | null; targetAssetId?: string | null }, a: string) =>
  (c.sourceAssetId === a ? c.targetAssetId : c.sourceAssetId) ?? null;
const isOpgw = (c: { sourceRole?: string | null; targetRole?: string | null }) => c.sourceRole === 'IN' && c.targetRole === 'IN';

export function projectTrace(seedCableId: string, graph: TraceGraph): TraceProjection | null {
  const cableById = new Map(graph.cables.map((c) => [c.id, c]));
  const seed = cableById.get(seedCableId);
  if (!seed) return null;
  const kindOf = (id: string | null | undefined) => (id ? (graph.assets.find((a) => a.id === id)?.connectionKind ?? null) : null);

  // start = passive 끝점 우선
  const src = seed.sourceAssetId ?? null;
  const tgt = seed.targetAssetId ?? null;
  const start = (kindOf(src) === null ? src : kindOf(tgt) === null ? tgt : src);
  if (!start) return null;
  const cableType = seed.cableType ?? 'FIBER';

  const tr = cableTrace(start, cableType, graph.assets, graph.cables);
  const tracedCables = tr.cableIds.map((id) => cableById.get(id)).filter(Boolean) as typeof graph.cables;

  // 코어번호 K
  const K = seed.number ?? tracedCables.find((c) => roleAt(c, c.sourceAssetId === start ? start : (c.sourceAssetId ?? '')) === 'OUT')?.number ?? null;

  const collapse = (id: string | null | undefined): string | null => {
    if (!id) return null;
    if (kindOf(id) === 'conduit') return graph.parentById.get(id) ?? id;
    return id;
  };

  // ── 토폴로지 nodes ──
  const nodeById = new Map<string, TraceNode>();
  for (const raw of tr.nodeIds) {
    const id = collapse(raw)!;
    if (nodeById.has(id)) continue;
    const code = graph.codeById.get(id) ?? null;
    nodeById.set(id, {
      nodeId: id,
      nodeName: graph.nameById.get(id) ?? id,
      substationName: graph.subNameById.get(id) ?? '',
      substationId: '',
      floorId: null,
      materialCategoryCode: code === 'OFD' ? 'EQP-OFD' : null,
      isSource: collapse(start) === id,
      isTarget: false,
    });
  }

  // ── 토폴로지 edges ──
  const edges: TraceEdge[] = [];
  let seedFiberEdgeId: string | null = isOpgw(seed) ? seed.id : null;
  for (const c of tracedCables) {
    const s = collapse(c.sourceAssetId);
    const t = collapse(c.targetAssetId);
    if (!s || !t || s === t) continue;
    if (isOpgw(c)) {
      edges.push({ id: c.id, sourceAssetId: s, targetAssetId: t, type: 'fiberPath', fiberPathId: c.id, fiberPortNumber: K ?? undefined });
      // 시드가 OUT 이면 그 슬롯에 붙은 OPGW 를 seed 광edge 로
      if (!seedFiberEdgeId && (c.sourceAssetId === seed.sourceAssetId || c.targetAssetId === seed.targetAssetId
        || c.sourceAssetId === seed.targetAssetId || c.targetAssetId === seed.sourceAssetId)) seedFiberEdgeId = c.id;
    } else {
      edges.push({ id: c.id, sourceAssetId: s, targetAssetId: t, type: 'cable' });
    }
  }

  // ── rings (광 edge 만) ──
  const fiberEdges = edges.filter((e) => e.type === 'fiberPath');
  const adjacency = new Map<string, Set<string>>();
  const edgeMap = new Map<string, TraceEdge>();
  for (const e of fiberEdges) {
    edgeMap.set(e.id, e);
    if (!adjacency.has(e.sourceAssetId)) adjacency.set(e.sourceAssetId, new Set());
    if (!adjacency.has(e.targetAssetId)) adjacency.set(e.targetAssetId, new Set());
    adjacency.get(e.sourceAssetId)!.add(e.targetAssetId);
    adjacency.get(e.targetAssetId)!.add(e.sourceAssetId);
  }
  const rings = detectRings(adjacency, edgeMap, nodeById);

  // ── steps (경로상세) : start→최원거리 BFS 메인 체인, 슬롯→OFD 접힘 ──
  const steps = buildSteps(start, tr.nodeIds, tracedCables, graph, collapse, kindOf);

  return {
    seedCableId, nodeIds: tr.nodeIds, cableIds: tr.cableIds,
    steps, nodes: [...nodeById.values()], edges, rings, seedFiberEdgeId, truncated: !!tr.truncated,
  };
}

function buildSteps(
  start: string,
  nodeIds: string[],
  tracedCables: { id: string; sourceAssetId?: string | null; targetAssetId?: string | null; sourceRole?: string | null; targetRole?: string | null }[],
  graph: TraceGraph,
  collapse: (id: string | null | undefined) => string | null,
  kindOf: (id: string | null | undefined) => string | null,
): TraceStep[] {
  // 인접 + BFS parent(서브그래프)
  const adj = new Map<string, { to: string; cable: typeof tracedCables[number] }[]>();
  const node = new Set(nodeIds);
  for (const c of tracedCables) {
    const a = c.sourceAssetId, b = c.targetAssetId;
    if (!a || !b || !node.has(a) || !node.has(b)) continue;
    (adj.get(a) ?? adj.set(a, []).get(a)!).push({ to: b, cable: c });
    (adj.get(b) ?? adj.set(b, []).get(b)!).push({ to: a, cable: c });
  }
  const parent = new Map<string, { from: string; cable: typeof tracedCables[number] } | null>([[start, null]]);
  const order: string[] = [start];
  const q = [start];
  let far = start;
  while (q.length) {
    const cur = q.shift()!;
    far = cur;
    for (const { to, cable } of adj.get(cur) ?? []) {
      if (parent.has(to)) continue;
      parent.set(to, { from: cur, cable });
      order.push(to);
      q.push(to);
    }
  }
  // start→far 역추적 체인
  const chain: { id: string; cable: typeof tracedCables[number] | null }[] = [];
  let cur: string | null = far;
  while (cur) {
    const p = parent.get(cur) ?? null;
    chain.unshift({ id: cur, cable: p?.cable ?? null });
    cur = p?.from ?? null;
  }
  // 슬롯→OFD 접기(연속 중복 제거) + step 생성
  const steps: TraceStep[] = [];
  for (let i = 0; i < chain.length; i++) {
    const cid = collapse(chain[i].id)!;
    if (steps.length && steps[steps.length - 1].id === cid) continue; // 슬롯 접힘 중복
    const incoming = chain[i].cable;
    const isFiberEdge = !!incoming && incoming.sourceRole === 'IN' && incoming.targetRole === 'IN';
    const isOfd = graph.codeById.get(cid) === 'OFD';
    steps.push({
      id: cid,
      label: isOfd ? (graph.subNameById.get(cid) ?? graph.nameById.get(cid) ?? cid) : (graph.nameById.get(cid) ?? cid),
      isEndpoint: i === 0 || i === chain.length - 1,
      isFiberEdge,
    });
  }
  return steps;
}
```
> `adj.get(a) ?? adj.set(a,[]).get(a)!` 패턴이 lint 에 걸리면 명시적 분기로 푼다. `detectRings` 의 nodeMap 은 `Map<string,TraceNode>` 면 됨(여기선 nodeById 그대로).
