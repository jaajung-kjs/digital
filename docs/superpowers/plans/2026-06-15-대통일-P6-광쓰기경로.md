# 연결 모델 대통일 — P6 광 쓰기 경로 마이그레이션 (클린 커오버) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development. Steps use checkbox (`- [ ]`).

**Goal:** 광 연결의 *쓰기* 경로를 신모델(슬롯/OPGW/OUT)로 전환한다 — 경로 관리자(OFD↔OFD = 슬롯2+OPGW), SlotCorePicker(설비↔슬롯 OUT 코어), ConnectionDiagram 신모델 읽기 — 그리고 P4가 만든 옛 중복 표현(옛 광케이블·FiberPath)을 `__fromCableId` 마커로 정밀 제거해 **신모델을 유일 진실로** 만든다. 백엔드 모델/스키마/컬럼 물리 삭제는 P7.

**Architecture:** 신 쓰기 헬퍼(코드→assetTypeId, 라우트/코어 페이로드 빌더, 순수+테스트) → 경로 관리자 재작성(FiberPathManager 대체) → SlotCorePicker 재작성(OfdPortPicker 대체, 캔버스 endpoint=슬롯·number=코어·role OUT 주입) → ConnectionDiagram 신모델 → 일회성 커오버 스크립트(`planFiberCutover` 순수 + 실행/revert) → 죽은 헬퍼 삭제·검증.

**Tech Stack:** React/TS/Zustand, React Query, Node/Prisma, Vitest.

**참조 spec:** `docs/superpowers/specs/2026-06-15-대통일-P6-광쓰기경로-design.md`. 재사용: `traceGraph`(useTraceGraph/remoteSlotSubstation), `slotRegister.buildSlotCoreRows`, P4 `scripts/migrateFiberToSlots.ts`(커오버 스크립트 템플릿).

```ts
import { useAssetTypes } from './useAssetTypes';

/** assetType code → id (placementKind 없는 종류, 예: 'OFD-SLOT' 조회용). */
export function useAssetTypeIdByCode(code: string): string | undefined {
  const { data: types = [] } = useAssetTypes();
  return useMemo(() => types.find((t) => t.code === code)?.id, [types, code]);
}
```

### Step 5: 구현 — Create `frontend/src/features/fiber/fiberWrite.ts`:
```ts
interface OfdRef { id: string; substationId: string; substationName: string | null }
interface CatRef { id: string; code: string; name: string; displayColor: string | null }

export interface SlotCreate {
  id: string; substationId: string; assetTypeId: string;
  assetType: { code: string; connectionKind: 'conduit'; placementKind: null };
  name: string; parentAssetId: string;
  floorId: null; positionX: null; positionY: null; width2d: null; height2d: null;
  rotation: number; totalU: null; slotIndex: null; slotSpan: null;
  description: null; manager: null; installDate: null; status: null;
  warrantyUntil: null; replaceDue: null; sourcePresetId: null; sortOrder: number; updatedAt: string;
}
export interface CableCreate {
  id: string; sourceAssetId: string; targetAssetId: string; cableType: 'FIBER';
  categoryId: string; categoryCode: string; categoryName: string; displayColor: string | null;
  sourceRole: 'IN' | 'OUT' | null; targetRole: 'IN' | 'OUT' | null; number: number | null;
  specParams: Record<string, unknown>; specification: string;
  pathPoints: [number, number][] | null; pathLength: number | null; bufferLength: number; totalLength: number | null;
}

const baseSlot = (id: string, ofd: OfdRef, remoteName: string, slotTypeId: string): SlotCreate => ({
  id, substationId: ofd.substationId, assetTypeId: slotTypeId,
  assetType: { code: 'OFD-SLOT', connectionKind: 'conduit', placementKind: null },
  name: remoteName, parentAssetId: ofd.id,
  floorId: null, positionX: null, positionY: null, width2d: null, height2d: null,
  rotation: 0, totalU: null, slotIndex: null, slotSpan: null,
  description: null, manager: null, installDate: null, status: null,
  warrantyUntil: null, replaceDue: null, sourcePresetId: null, sortOrder: 0, updatedAt: '',
});

/** OFD↔OFD 경로 = 슬롯2(각 OFD 자식, 대국명) + OPGW(IN-IN, cores). */
export function buildRouteCreate(p: {
  localOfd: OfdRef; remoteOfd: OfdRef; cores: number; slotTypeId: string; opgwCategory: CatRef;
  ids: { slotA: string; slotB: string; opgw: string };
}): { slots: SlotCreate[]; opgw: CableCreate } {
  const slotA = baseSlot(p.ids.slotA, p.localOfd, p.remoteOfd.substationName ?? '대국', p.slotTypeId);
  const slotB = baseSlot(p.ids.slotB, p.remoteOfd, p.localOfd.substationName ?? '대국', p.slotTypeId);
  const opgw: CableCreate = {
    id: p.ids.opgw, sourceAssetId: p.ids.slotA, targetAssetId: p.ids.slotB, cableType: 'FIBER',
    categoryId: p.opgwCategory.id, categoryCode: p.opgwCategory.code, categoryName: p.opgwCategory.name,
    displayColor: p.opgwCategory.displayColor, sourceRole: 'IN', targetRole: 'IN', number: null,
    specParams: { cores: p.cores }, specification: p.opgwCategory.name,
    pathPoints: null, pathLength: null, bufferLength: 4, totalLength: null,
  };
  return { slots: [slotA, slotB], opgw };
}

/** 설비↔슬롯 OUT 코어 케이블. 슬롯 끝 = OUT. */
export function buildCoreOutCable(p: {
  id: string; equipmentId: string; slotId: string; coreNumber: number; category: CatRef;
  pathPoints: [number, number][] | null; pathLength: number | null; bufferLength: number; totalLength: number | null;
}): CableCreate {
  return {
    id: p.id, sourceAssetId: p.equipmentId, targetAssetId: p.slotId, cableType: 'FIBER',
    categoryId: p.category.id, categoryCode: p.category.code, categoryName: p.category.name,
    displayColor: p.category.displayColor, sourceRole: null, targetRole: 'OUT', number: p.coreNumber,
    specParams: {}, specification: p.category.name,
    pathPoints: p.pathPoints, pathLength: p.pathLength, bufferLength: p.bufferLength, totalLength: p.totalLength,
  };
}

/** 점유 코어번호 + 용량 → 가장 작은 빈 번호(없으면 용량+1로 확장). */
export function nextFreeCore(occupied: number[], cores: number): number {
  const set = new Set(occupied);
  for (let n = 1; n <= cores; n++) if (!set.has(n)) return n;
  return cores + 1;
}

interface CableLike { id: string; sourceAssetId?: string | null; targetAssetId?: string | null; sourceRole?: string | null; targetRole?: string | null; cableType?: string | null }
/** 경로 삭제 = 두 슬롯 + 그 슬롯에 닿는 모든 FIBER 케이블(OPGW+OUT). */
export function routeDeleteIds(slotAId: string, slotBId: string, cables: CableLike[]): { assetIds: string[]; cableIds: string[] } {
  const slots = new Set([slotAId, slotBId]);
  const cableIds = cables
    .filter((c) => c.cableType === 'FIBER' && ((c.sourceAssetId && slots.has(c.sourceAssetId)) || (c.targetAssetId && slots.has(c.targetAssetId))))
    .map((c) => c.id);
  return { assetIds: [slotAId, slotBId], cableIds };
}
```

### Step 7: 경로 관리자 — Create `frontend/src/features/fiber/components/FiberRouteManager.tsx`.
OFD 의 슬롯(경로) 목록 + "경로 추가"(대국 OFD 선택 + 코어수) + 삭제. 슬롯 발견은 OfdFiberRegister 패턴(`assets.filter(parentAssetId===ofdId && connectionKind==='conduit')`), 대국명/용량은 `remoteSlotSubstation`+OPGW specParams.cores, 생성/삭제는 `buildRouteCreate`/`routeDeleteIds`+`put`/`remove`.
```tsx
import { useMemo, useState } from 'react';
import { useEffectiveAssets, useEffectiveCables } from '../../workingCopy/hooks';
import { useSubstationWorkingCopy } from '../../workingCopy/substationStore';
import { useTraceGraph, remoteSlotSubstation } from '../../trace/traceGraph';
import { useAssetTypeIdByCode } from '../../assets/useAssetTypeIdByCode';
import { useCableCategories } from '../../cables/hooks/useCableCategories';
import { generateTempId } from '../../../utils/idHelpers';
import { buildRouteCreate, routeDeleteIds } from '../fiberWrite';
import type { Asset } from '../../../types/asset';

export function FiberRouteManager({ ofdId }: { ofdId: string }) {
  const assets = useEffectiveAssets() as Asset[];
  const cables = useEffectiveCables();
  const { graph } = useTraceGraph();
  const slotTypeId = useAssetTypeIdByCode('OFD-SLOT');
  const { data: categories = [] } = useCableCategories();
  const opgwCat = categories.find((c) => c.code === 'CBL-OPGW');
  const [adding, setAdding] = useState(false);
  const [cores, setCores] = useState(24);

  const ofd = assets.find((a) => a.id === ofdId);
  const slots = useMemo(() => assets.filter((a) => a.parentAssetId === ofdId && a.assetType?.connectionKind === 'conduit'), [assets, ofdId]);
  const ofds = useMemo(() => assets.filter((a) => a.assetType?.code === 'OFD' && a.id !== ofdId), [assets, ofdId]);

  const addRoute = (remote: Asset) => {
    if (!slotTypeId || !opgwCat || !ofd) return;
    const { slots: newSlots, opgw } = buildRouteCreate({
      localOfd: { id: ofd.id, substationId: ofd.substationId, substationName: graph?.subNameById.get(ofd.id) ?? null },
      remoteOfd: { id: remote.id, substationId: remote.substationId, substationName: graph?.subNameById.get(remote.id) ?? null },
      cores, slotTypeId, opgwCategory: opgwCat,
      ids: { slotA: generateTempId(), slotB: generateTempId(), opgw: generateTempId() },
    });
    const wc = useSubstationWorkingCopy.getState();
    for (const s of newSlots) wc.put('assets', s);
    wc.put('cables', opgw);
    setAdding(false);
  };

  const deleteRoute = (slot: Asset) => {
    // 대국 슬롯(twin)도 함께 — OPGW 의 반대 끝.
    const wc = useSubstationWorkingCopy.getState();
    const opgw = cables.find((c) => (c as { sourceRole?: string }).sourceRole === 'IN' && (c as { targetRole?: string }).targetRole === 'IN'
      && ((c as { sourceAssetId?: string }).sourceAssetId === slot.id || (c as { targetAssetId?: string }).targetAssetId === slot.id));
    const twinId = opgw ? ((opgw as { sourceAssetId?: string }).sourceAssetId === slot.id ? (opgw as { targetAssetId?: string }).targetAssetId : (opgw as { sourceAssetId?: string }).sourceAssetId) : undefined;
    if (!twinId) return;
    const { assetIds, cableIds } = routeDeleteIds(slot.id, twinId, cables as never[]);
    if (!confirm(`이 경로와 코어 연결 ${Math.max(0, cableIds.length - 1)}개를 삭제합니다.`)) return;
    for (const id of cableIds) wc.remove('cables', id);
    for (const id of assetIds) wc.remove('assets', id);
  };

  return (
    <div className="space-y-2">
      <div className="flex items-center justify-between px-1">
        <h4 className="text-xs font-semibold text-content">광 경로</h4>
        <button className="text-xs text-primary" onClick={() => setAdding((v) => !v)} disabled={!slotTypeId || !opgwCat}>+ 경로 추가</button>
      </div>
      {adding && (
        <div className="rounded border border-line p-2 space-y-2">
          <label className="text-[11px] text-content-muted flex items-center gap-1">코어 수
            <input type="number" min={1} value={cores} onChange={(e) => setCores(Math.max(1, Number(e.target.value) || 1))} className="w-16 border border-line rounded px-1 text-xs" />
          </label>
          <div className="max-h-40 overflow-auto space-y-0.5">
            {ofds.map((o) => (
              <button key={o.id} onClick={() => addRoute(o)} className="block w-full text-left text-xs px-1.5 py-1 rounded hover:bg-surface-2">
                {graph?.subNameById.get(o.id) ?? o.name} 대국
              </button>
            ))}
          </div>
        </div>
      )}
      <table className="w-full border-collapse">
        <tbody>
          {slots.map((slot) => {
            const remote = graph ? remoteSlotSubstation(slot.id, graph) : null;
            return (
              <tr key={slot.id} className="border-b border-line text-[13px]">
                <td className="px-2 py-1.5 text-content">{remote ?? slot.name}</td>
                <td className="px-2 py-1.5 text-right">
                  <button className="text-xs text-danger" onClick={() => deleteRoute(slot)}>삭제</button>
                </td>
              </tr>
            );
          })}
          {slots.length === 0 && <tr><td className="px-2 py-2 text-xs text-content-faint">경로 없음 — "경로 추가"로 대국 트렁크 생성</td></tr>}
        </tbody>
      </table>
    </div>
  );
}
```

```tsx
import { useMemo, useState } from 'react';
import { useEffectiveAssets, useEffectiveCables } from '../../workingCopy/hooks';
import { useTraceGraph, remoteSlotSubstation } from '../../trace/traceGraph';
import { buildSlotCoreRows } from '../../fiber/slotRegister';
import { nextFreeCore } from '../../fiber/fiberWrite';
import type { Asset } from '../../../types/asset';

export function SlotCorePicker({ ofdId, onSelect, onCancel }: {
  ofdId: string; onSelect: (sel: { slotId: string; coreNumber: number }) => void; onCancel: () => void;
}) {
  const assets = useEffectiveAssets() as Asset[];
  const cables = useEffectiveCables();
  const { graph } = useTraceGraph();
  const slots = useMemo(() => assets.filter((a) => a.parentAssetId === ofdId && a.assetType?.connectionKind === 'conduit'), [assets, ofdId]);
  const [slotId, setSlotId] = useState<string | null>(slots[0]?.id ?? null);

  const rows = useMemo(() => {
    const slot = slots.find((s) => s.id === slotId);
    return slot ? buildSlotCoreRows(slot as never, cables as never[], graph) : [];
  }, [slots, slotId, cables, graph]);
  const occupied = rows.filter((r) => r.occupied).map((r) => r.coreNumber);
  const capacity = rows.length || 24;
  const suggested = nextFreeCore(occupied, capacity);

  if (!slots.length) return (
    <div className="p-3 text-sm text-content-faint">이 OFD 에 광 경로가 없습니다. OFD 상세의 "경로 추가"로 대국 트렁크를 먼저 만드세요.
      <div className="mt-2"><button className="text-xs text-primary" onClick={onCancel}>닫기</button></div>
    </div>
  );

  return (
    <div className="p-3 space-y-2 min-w-[280px]">
      <div className="flex flex-wrap gap-1">
        {slots.map((s) => (
          <button key={s.id} onClick={() => setSlotId(s.id)}
            className={`text-xs px-2 py-1 rounded ${slotId === s.id ? 'bg-primary text-white' : 'bg-surface-2 text-content-muted'}`}>
            {(graph && remoteSlotSubstation(s.id, graph)) ?? s.name}
          </button>
        ))}
      </div>
      <div className="grid grid-cols-8 gap-1">
        {rows.map((r) => (
          <button key={r.coreNumber} disabled={r.occupied}
            onClick={() => slotId && onSelect({ slotId, coreNumber: r.coreNumber })}
            className={`text-[11px] h-7 rounded tabular-nums ${
              r.occupied ? 'bg-surface-3 text-content-faint cursor-not-allowed'
              : r.coreNumber === suggested ? 'bg-info-bg text-primary ring-1 ring-primary' : 'bg-surface-2 hover:bg-surface-3'}`}>
            {r.coreNumber}
          </button>
        ))}
      </div>
      <div className="text-right"><button className="text-xs text-content-faint" onClick={onCancel}>취소</button></div>
    </div>
  );
}
```

### Step 3: 구현 — Create `backend/src/migration/fiberCutover.ts`:
```ts
interface OutCable { id: string; specParams: unknown }
interface FiberPathRow { id: string }

/** 커오버 계획: 마이그레이션 OUT 의 __fromCableId 옛 광케이블 + 전 FiberPath 삭제. 순수. */
export function planFiberCutover(migratedOut: OutCable[], fiberPaths: FiberPathRow[]): { oldCableIds: string[]; fiberPathIds: string[] } {
  const oldCableIds = new Set<string>();
  for (const c of migratedOut) {
    const sp = (c.specParams ?? {}) as Record<string, unknown>;
    if (sp.__migration === 'fiberToSlots' && typeof sp.__fromCableId === 'string') oldCableIds.add(sp.__fromCableId);
  }
  return { oldCableIds: [...oldCableIds], fiberPathIds: fiberPaths.map((f) => f.id) };
}
```

```ts
// 핵심 골격 (P4 러너 양식 준수)
import prisma from '../src/config/prisma.js';
import { planFiberCutover } from '../src/migration/fiberCutover.js';

const dryRun = process.argv.includes('--dry-run');
async function main() {
  const migratedOut = await prisma.cable.findMany({
    where: { specParams: { path: ['__migration'], equals: 'fiberToSlots' } },
    select: { id: true, specParams: true },
  });
  const fiberPaths = await prisma.fiberPath.findMany({ select: { id: true } });
  const plan = planFiberCutover(migratedOut, fiberPaths);
  // 안전: 삭제 대상 옛 케이블이 실제 존재·fiberPathId 보유(마이그레이션 출처)인지 교차확인
  const existing = await prisma.cable.findMany({ where: { id: { in: plan.oldCableIds } }, select: { id: true, fiberPathId: true } });
  console.log(`커오버 계획 — 옛 광케이블 ${existing.length}/${plan.oldCableIds.length} · FiberPath ${plan.fiberPathIds.length}`);
  if (dryRun) { console.log('dry-run — 삭제 안 함'); return; }
  await prisma.$transaction(async (tx) => {
    await tx.cable.deleteMany({ where: { id: { in: plan.oldCableIds } } });
    await tx.fiberPath.deleteMany({ where: { id: { in: plan.fiberPathIds } } });
  });
  console.log('커오버 완료');
}
main().then(() => process.exit(0)).catch((e) => { console.error(e); process.exit(1); });
```
