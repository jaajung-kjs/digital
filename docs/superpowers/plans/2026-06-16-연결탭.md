```ts
// src/features/connections/hooks/useAssetConnections.ts
import { useMemo } from 'react';
import { useEffectiveAssets } from '../../workingCopy/hooks';
import { useTraceGraph, type TraceGraph } from '../../trace/traceGraph';
import { projectTrace, type TraceProjection } from '../../trace/traceProjection';
import { buildSelfSideChecker, buildEndpointNameResolver } from '../endpointName';
import { useCableCategories } from '../../cables/hooks/useCableCategories';
import type { Asset } from '../../../types/asset';

export interface ConnectionRow { cableId: string; fromName: string; toName: string; truncated: boolean }
export interface ConnectionGroup { key: string; label: string; color: string | null; rows: ConnectionRow[] }

interface CatGroup { key: string; label: string; color: string | null }

/** 순수 파생 — 테스트/재사용 위해 projectTrace·카테고리 해소를 주입받는다. */
export function buildConnectionGroups(args: {
  graph: TraceGraph;
  assets: Asset[];
  assetId: string;
  projectFn: (seedCableId: string, graph: TraceGraph) => TraceProjection | null;
  categoryGroupOf: (cable: TraceGraph['cables'][number]) => CatGroup;
}): ConnectionGroup[] {
  const { graph, assets, assetId, projectFn, categoryGroupOf } = args;
  const isSelf = buildSelfSideChecker(assets, assetId);
  const nameOf = buildEndpointNameResolver(assets);
  const kindById = new Map(graph.assets.map((a) => [a.id, a.connectionKind ?? null]));

  // self set 에 닿는 케이블만.
  const candidates = graph.cables.filter((c) => isSelf(c.sourceAssetId) || isSelf(c.targetAssetId));

  const groupsByKey = new Map<string, ConnectionGroup>();
  const seenSig = new Set<string>(); // 링 중복제거: 동일 nodeIds → 1행.

  for (const c of candidates) {
    const proj = projectFn(c.id, graph);
    if (!proj) continue;
    const sig = [...proj.nodeIds].sort().join('|');
    if (seenSig.has(sig)) continue;
    seenSig.add(sig);

    const selfEndpoint = isSelf(c.sourceAssetId) ? c.sourceAssetId! : c.targetAssetId!;
    const fromName = nameOf(selfEndpoint);
    // self 가 conduit(슬롯)면 step 은 부모 OFD 로 collapse 되어 있다.
    const selfStepId = kindById.get(selfEndpoint) === 'conduit'
      ? (graph.parentById.get(selfEndpoint) ?? selfEndpoint)
      : selfEndpoint;
    const endpoints = proj.steps.filter((s) => s.isEndpoint);
    const far = endpoints.find((s) => s.id !== selfStepId) ?? endpoints[endpoints.length - 1] ?? null;
    const toName = far?.label ?? '';

    const cg = categoryGroupOf(c);
    let g = groupsByKey.get(cg.key);
    if (!g) { g = { key: cg.key, label: cg.label, color: cg.color, rows: [] }; groupsByKey.set(cg.key, g); }
    g.rows.push({ cableId: c.id, fromName, toName, truncated: !!proj.truncated });
  }

  const groups = [...groupsByKey.values()];
  for (const g of groups) g.rows.sort((a, b) => a.fromName.localeCompare(b.fromName) || a.toName.localeCompare(b.toName));
  groups.sort((a, b) => a.label.localeCompare(b.label));
  return groups;
}

/** 케이블 종류 그룹(전원/네트워크/광 …) 해소 — categoryId→displayGroup, 없으면 cableType 폴백. */
function makeCategoryGroupOf(categories: { id: string; displayGroup: string | null; displayColor: string | null }[]) {
  const byId = new Map(categories.map((c) => [c.id, c]));
  return (cable: TraceGraph['cables'][number]): CatGroup => {
    const cat = cable.categoryId ? byId.get(cable.categoryId) : undefined;
    const key = cat?.displayGroup ?? cable.cableType ?? '기타';
    return { key, label: cat?.displayGroup ?? cable.cableType ?? '기타', color: cat?.displayColor ?? cable.displayColor ?? null };
  };
}

/** 연결 탭 데이터 훅 — 자산+보유자산의 종류별 연결(출발→도착, 링 중복제거). */
export function useAssetConnections(assetId: string): ConnectionGroup[] {
  const assets = useEffectiveAssets() as Asset[];
  const { graph } = useTraceGraph();
  const { data: categories = [] } = useCableCategories();
  return useMemo(() => {
    if (!graph) return [];
    return buildConnectionGroups({
      graph, assets, assetId,
      projectFn: projectTrace,
      categoryGroupOf: makeCategoryGroupOf(categories),
    });
  }, [graph, assets, assetId, categories]);
}
```
