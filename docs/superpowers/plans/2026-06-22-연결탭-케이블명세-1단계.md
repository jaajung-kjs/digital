# 연결탭 재설계 — 1단계: 케이블 명세 리스트 + 선택 편집 카드

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development 또는 superpowers:executing-plans 로 task 단위 실행. 단계는 체크박스(`- [ ]`).

**Goal:** 자산 연결탭을 "토폴로지 트리"에서 "이 자산(+자식)에 직접 닿는 케이블 명세 리스트(종류별, 피더는 IN/OUT 중첩) + 행 클릭 시 하단 고정 DetailCard 속성편집"으로 전환한다.

**Architecture:** 리스트는 회로추적(`buildConnectionDiagram`) 없이 `isSelf(끝점)` 케이블을 종류별로 모으는 순수 함수 `buildCableRegister`로 만든다. 도면 하이라이트는 기존 `setSelectedComponent → useSelectionHighlight` 파이프라인을 그대로 재사용한다. 경로 인라인뷰(세로 홉)·경로 알고리즘 정교화는 **2단계**로 분리한다.

**Tech Stack:** React + TypeScript, zustand(selectionStore), 공용 UI(`DetailCard`/`EditableField`), vitest.

## Global Constraints
- UI는 `frontend/src/components/ui/CONVENTIONS.md` 준수 — DetailCard 계열 재사용, 타이포 스케일(`text-xs` 미만 금지), 시맨틱 색 토큰만.
- 케이블 쓰기는 `stageCableUpdate`/`stageCableDelete`(SSOT)로만. specParams는 이 카드에서 편집 금지(선번장/계통 그리드 전담) — 안내만.
- `buildConnectionDiagram`/`DiagramTree`는 **삭제하지 않는다**: `useSelectionHighlight`(resolveHighlight)가 `useAssetDiagram`을 계속 사용한다. DiagramTree 렌더러는 1단계 후 탭에서 미사용이 되지만 2단계(경로뷰) 결정까지 보존.
- dev: DB만 docker + `npm run dev`. 검증은 `npm run build`(타입체크)/`vitest`.

```ts
import { roleAt } from '../cables/cableEndpoint';
import { buildSelfSideChecker, buildEndpointNameResolver } from './endpointName';
import { FEEDER_CODE } from '../assets/distributionSubtree';
import { toMapById } from '../../utils/byId';
import type { TraceGraph } from '../trace/traceGraph';
import type { Asset } from '../../types/asset';

type Cable = TraceGraph['cables'][number];
interface CategoryGroup { key: string; label: string; color: string | null }

export interface CableRow {
  cable: Cable; selfId: string; remoteId: string | null;
  fromName: string; toName: string; roleAtSelf: string | null; number: number | null;
}
export interface FeederGroup { feederId: string; feederName: string; inRow: CableRow | null; outRows: CableRow[]; }
export interface CableSection { key: string; label: string; color: string | null; feeders: FeederGroup[]; rows: CableRow[]; }

export function buildCableRegister(opts: {
  graph: TraceGraph; assets: Asset[]; assetId: string;
  categoryGroupOf: (c: { categoryId?: string|null; cableType?: string|null; displayColor?: string|null }) => CategoryGroup;
}): CableSection[] {
  const { graph, assets, assetId, categoryGroupOf } = opts;
  const isSelf = buildSelfSideChecker(assets, assetId);
  const nameOf = buildEndpointNameResolver(assets);
  const byId = toMapById(assets);
  const resolveName = (id?: string | null) => (id ? (nameOf(id) || graph.nameById.get(id) || id) : '');
  const isFeeder = (id?: string | null) => !!id && byId.get(id)?.assetType?.code === FEEDER_CODE;

  const toRow = (c: Cable): CableRow => {
    const selfId = (isSelf(c.sourceAssetId) ? c.sourceAssetId : c.targetAssetId) as string;
    const remoteId = selfId === c.sourceAssetId ? (c.targetAssetId ?? null) : (c.sourceAssetId ?? null);
    return { cable: c, selfId, remoteId, fromName: resolveName(c.sourceAssetId), toName: resolveName(c.targetAssetId), roleAtSelf: roleAt(c, selfId), number: c.number ?? null };
  };

  const sections = new Map<string, CableSection>();
  const feederMap = new Map<string, FeederGroup>();
  const sec = (cg: CategoryGroup) => {
    let s = sections.get(cg.key);
    if (!s) { s = { key: cg.key, label: cg.label, color: cg.color, feeders: [], rows: [] }; sections.set(cg.key, s); }
    else if (!s.color && cg.color) s.color = cg.color;
    return s;
  };

  for (const c of graph.cables) {
    if (!isSelf(c.sourceAssetId) && !isSelf(c.targetAssetId)) continue;
    const s = sec(categoryGroupOf(c));
    const row = toRow(c);
    const feederId = [c.sourceAssetId, c.targetAssetId].find((e) => isFeeder(e) && isSelf(e)) ?? null;
    if (feederId) {
      let fg = feederMap.get(feederId);
      if (!fg) { fg = { feederId, feederName: resolveName(feederId), inRow: null, outRows: [] }; feederMap.set(feederId, fg); s.feeders.push(fg); }
      if (roleAt(c, feederId) === 'IN') fg.inRow = row; else fg.outRows.push(row);
    } else {
      s.rows.push(row);
    }
  }

  const out = [...sections.values()];
  for (const s of out) {
    s.rows.sort((a, b) => a.toName.localeCompare(b.toName));
    s.feeders.sort((a, b) => a.feederName.localeCompare(b.feederName));
    for (const f of s.feeders) f.outRows.sort((a, b) => (a.number ?? 0) - (b.number ?? 0) || a.toName.localeCompare(b.toName));
  }
  out.sort((a, b) => a.label.localeCompare(b.label));
  return out;
}
```

```tsx
import { DetailCard, DetailCardHeader, DetailRow } from '../../../components/ui';
import { EditableField, type EditableFieldOption } from '../../assets/components/EditableField';
// ...(useEffectiveCables/useCableCategories/useTraceGraph/useSubstationWorkingCopy 기존 import 유지)

export function CableInspector({ cableId, onDeleted }: { cableId: string; onDeleted?: () => void }) {
  const cables = useEffectiveCables();
  const { data: categories } = useCableCategories();
  const { graph } = useTraceGraph();
  const cable = cables.find((c) => c.id === cableId);
  if (!cable) return <div className="text-xs text-content-faint">케이블을 찾을 수 없습니다.</div>;

  const patch = (p: Record<string, unknown>) => useSubstationWorkingCopy.getState().stageCableUpdate(cableId, p);
  const del = () => { useSubstationWorkingCopy.getState().stageCableDelete(cableId); onDeleted?.(); };

  const src = graph?.nameById.get(cable.sourceAssetId as string) ?? '';
  const tgt = graph?.nameById.get(cable.targetAssetId as string) ?? '';
  const categoryId = (cable.categoryId as string | null) ?? '';
  const categoryOptions: EditableFieldOption[] = [
    { value: '', label: '— 미분류 —' },
    ...(categories ?? []).filter((c) => c.isActive).map((c) => ({ value: c.id, label: c.name })),
  ];
  const commitCategory = (id: string) => {
    const c = (categories ?? []).find((x) => x.id === id) ?? null;
    patch({ categoryId: id || null, categoryName: c?.name ?? null, displayColor: c?.displayColor ?? null });
  };
  const hasSpec = cable.specParams != null && Object.keys(cable.specParams as object).length > 0;
  const isOpgw = (cable.specParams as { coreMeta?: unknown } | null)?.coreMeta != null;

  return (
    <DetailCard>
      <DetailCardHeader title={`${src} – ${tgt}`} onDelete={del} />
      <DetailRow label="종류">
        <EditableField value={categoryId} type="select" options={categoryOptions} ariaLabel="종류" valueClickEdits
          display={(v) => categoryOptions.find((o) => o.value === v)?.label ?? '— 미분류 —'} onCommit={commitCategory} />
      </DetailRow>
      <DetailRow label="번호">
        <EditableField value={cable.number != null ? String(cable.number) : ''} ariaLabel="번호" placeholder="번호" valueClickEdits
          onCommit={(v) => { const n = v.trim() ? parseInt(v, 10) : NaN; patch({ number: Number.isNaN(n) ? null : n }); }} />
      </DetailRow>
      <DetailRow label="라벨">
        <EditableField value={(cable.label as string | null) ?? ''} ariaLabel="라벨" placeholder="라벨" valueClickEdits onCommit={(v) => patch({ label: v.trim() || null })} />
      </DetailRow>
      <DetailRow label="색상">
        <EditableField value={(cable.color as string | null) ?? ''} ariaLabel="색상" placeholder="종류 색 상속" valueClickEdits onCommit={(v) => patch({ color: v.trim() || null })} />
      </DetailRow>
      <DetailRow label="설명">
        <EditableField value={(cable.description as string | null) ?? ''} ariaLabel="설명" placeholder="설명" valueClickEdits onCommit={(v) => patch({ description: v.trim() || null })} />
      </DetailRow>
      {hasSpec && <p className="text-xs text-content-faint mt-1">상세값은 {isOpgw ? '선번장' : '계통'} 그리드에서 편집합니다.</p>}
    </DetailCard>
  );
}
```

```tsx
import { useMemo } from 'react';
import { useEffectiveAssets } from '../../workingCopy/hooks';
import { useTraceGraph } from '../../trace/traceGraph';
import { useCableCategories } from '../../cables/hooks/useCableCategories';
import { makeCategoryGroupOf } from '../hooks/useAssetConnections';
import { useSelectionStore } from '../../workspace/selectionStore';
import { buildCableRegister, type CableRow } from '../cableRegister';
import { CableInspector } from '../../cables/components/CableInspector';

export function AssetConnectionsTab({ assetId }: { assetId: string }) {
  const assets = useEffectiveAssets();
  const { graph, isLoading } = useTraceGraph();
  const { data: categories } = useCableCategories();
  const selectedCableId = useSelectionStore((s) => s.selectedCableId);

  const sections = useMemo(() => {
    if (!graph || !categories) return [];
    return buildCableRegister({ graph, assets, assetId, categoryGroupOf: makeCategoryGroupOf(categories) });
  }, [graph, categories, assets, assetId]);

  if (isLoading) return <div className="py-8 text-center text-xs text-content-faint">불러오는 중…</div>;
  if (sections.length === 0) return <div className="py-8 text-center text-xs text-content-faint">연결된 케이블이 없습니다.</div>;

  // 행 클릭 = 그 케이블 정밀 선택(하단 카드) + 도면 하이라이트(기존 파이프라인).
  const pick = (r: CableRow) => useSelectionStore.getState().setSelectedComponent(assetId, r.number, r.cable.id);
  const Row = ({ r, indent }: { r: CableRow; indent?: boolean }) => {
    const on = selectedCableId === r.cable.id;
    return (
      <button type="button" onClick={() => pick(r)}
        className={`w-full text-left px-3 py-2 border-b border-line text-sm ${indent ? 'pl-7' : ''} ${on ? 'bg-info-bg/40' : 'hover:bg-surface-2'}`}>
        <span className="text-content">{r.fromName} <span className="text-content-faint">→</span> {r.toName}</span>
        <span className="ml-2 text-xs text-content-faint">{r.roleAtSelf ?? ''}{r.number != null ? ` ·#${r.number}` : ''}</span>
      </button>
    );
  };

  return (
    <div className="flex flex-col gap-0">
      {sections.map((s) => (
        <div key={s.key}>
          <div className="flex items-center gap-1.5 px-3 py-1.5 bg-surface-2 border-b border-line">
            {s.color && <span className="inline-block w-2 h-2 rounded-full" style={{ backgroundColor: s.color }} />}
            <span className="text-xs font-semibold text-content-muted">{s.label}</span>
            <span className="ml-auto text-xs text-content-faint">{s.rows.length + s.feeders.reduce((n, f) => n + (f.inRow ? 1 : 0) + f.outRows.length, 0)}</span>
          </div>
          {s.feeders.map((f) => (
            <div key={f.feederId}>
              {f.inRow && <Row r={f.inRow} />}
              {f.outRows.map((r) => <Row key={r.cable.id} r={r} indent />)}
            </div>
          ))}
          {s.rows.map((r) => <Row key={r.cable.id} r={r} />)}
        </div>
      ))}

      {/* 선택한 케이블 — 하단 고정(sticky) 속성 카드 */}
      {selectedCableId && (
        <div className="sticky bottom-0 z-sticky bg-surface border-t-2 border-line p-3">
          <div className="mb-2 text-xs font-semibold text-content-muted">선택한 케이블</div>
          <CableInspector cableId={selectedCableId} onDeleted={() => useSelectionStore.getState().setSelectedAssetId(assetId)} />
        </div>
      )}
    </div>
  );
}
```
> 하단 고정은 탭 본문이 이미 `overflow-y-auto`(DetailTabs)이므로 `sticky bottom-0`로 구현(별도 레이아웃 개편 불필요). 삭제 후엔 선택 해제(`setSelectedAssetId(assetId)`가 selectedCableId 를 비움).
